from reamber.osu.OsuMap import OsuMap
import matplotlib.pyplot as plt

from reamber.algorithms.analysis.generic.rollingDensity import rollingDensity
import os

m = OsuMap.readFile("path/to/file.osu")

density = m.notes.hits().rollingDensity(window=5000, stride=2500)
plt.plot(list(density.keys()), list(density.values()))
plt.show()

activity(lastOffset: Optional[float] = None)¶

Calculates how long each Timed Object is active. Implicitly sorts object by offset

For example:

The algorithm calculates this:

SEC 1   2   3   4   5   6   7   8   9
BPM 100 ------> 200 --> 300 -------->

returns [(Timed<1>, 3000), (Timed<2>, 2000), (Timed<3>, 3000)]

Parameters

lastOffset – Last offset, if None, uses Timed.lastOffset()

Returns

A List of Tuples in the format [(Timed, Activity In ms), …]
