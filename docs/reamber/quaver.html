3. Multiply all Bpms and Svs by 1.5¶

from reamber.quaver.QuaMapObj import QuaMapObj

q = QuaMapObj()
q.readFile("file.qua")

for sv in q.svs.data():
    sv.multiplier *= 1.5

for bpm in q.bpms.data():
    bpm.bpm *= 1.5
